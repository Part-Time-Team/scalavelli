32     override def gameStateUpdated(gameState: PlayerGameState): Unit = {
33 
34       storeOpt match {

176   private def updateHistoryAndNotify(updatedState: PlayerGameState): Unit = {
177     this.updateHistory(updatedState)

181   private def updateHistory(newState: PlayerGameState): Unit = {

190   private def generateClientGameState(state: PlayerGameState,
191                                       turnHistory: History[PlayerGameState]): ClientGameState =
192     ClientGameState(
193       playerGameState = state,
194       canUndo = turnHistory.canPrevious,
195       canRedo = turnHistory.canNext,
196       canReset = turnHistory.canPrevious,
197       canDrawCard = !isTurnValid(state)
198     )
199 
200   /**
201     * Execute the history method, updates the history and the resulting state
202     *
203     * @param historyUpdate History method to execute.
204     */
205   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
206     val (optState, updatedHistory) = historyUpdate()
207     this.turnHistory = updatedHistory
208     if (optState.isDefined) {
209       this.storeOpt.get.onStateChanged(optState.get)
210       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
211     }
212   }
213 
214   /**
215     * Return if Turn is valid or not.
216     *
217     * @param currentState Current Game State.
218     * @return True if the Turn isn't valid, false anywhere.
219     */
220   private def isTurnValid(currentState: PlayerGameState) = {
221     val startState = turnHistory.headOption
222     startState.isDefined && gameInterface.validateTurn(
223       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
224   }
