192   private def generateClientGameState(state: PlayerGameState,
193                                       turnHistory: History[PlayerGameState]): ClientGameState =
194     ClientGameState(
195       playerGameState = state,
196       canUndo = turnHistory.canPrevious,
197       canRedo = turnHistory.canNext,
198       canReset = turnHistory.canPrevious,
199       canDrawCard = !isTurnValid(state)

202   /**
203    * Execute the history method, updates the history and the resulting state
204    *

206    */
207   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
208     val (optState, updatedHistory) = historyUpdate()
209     this.turnHistory = updatedHistory
210     if (optState.isDefined) {
211       this.storeOpt.get.onStateChanged(optState.get)
212       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
213     }
214   }

216   /**
217    * Return if Turn is valid or not.

220    * @return True if the Turn isn't valid, false anywhere.

222   private def isTurnValid(currentState: PlayerGameState) = {
223     val startState = turnHistory.headOption
