33     override def gameStateUpdated(gameState: PlayerGameState): Unit = {
34 
35       storeOpt match {

37           store.onStateChanged(gameState), turnHistory)))

138   private def makeCombination(cards: Seq[Card]): Unit = {
139     withState { state =>
140       this.gameInterface.playCombination(state.hand, state.board, cards) match {
141         case Right((updatedHand, updatedBoard)) =>
142           val updatedState = storeOpt.get.onLocalTurnStateChanged(updatedHand, updatedBoard)

145         case Left(error) => this.notifyEvent(ErrorEvent(error))

228   private def generateClientGameState(state: PlayerGameState,
229                                       turnHistory: History[PlayerGameState]): ClientGameState =
230     ClientGameState(

238   /**
239    * Execute the history method, updates the history and the resulting state
240    *
241    * @param historyUpdate History method to execute.
242    */
243   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
244     val (optState, updatedHistory) = historyUpdate()
245     this.turnHistory = updatedHistory
246     if (optState.isDefined) {
247       this.storeOpt.get.onStateChanged(optState.get)
248       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
249     }
250   }

252   /**
253    * Return if Turn is valid or not.
254    *
255    * @param currentState Current Game State.
256    * @return True if the Turn isn't valid, false anywhere.

258   private def isTurnValid(currentState: PlayerGameState) = {
259     val startState = turnHistory.headOption

261       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
262   }
