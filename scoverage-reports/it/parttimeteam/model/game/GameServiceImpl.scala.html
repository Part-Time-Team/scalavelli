33     override def gameStateUpdated(gameState: PlayerGameState): Unit = {
34 
35       storeOpt match {

137   private def makeCombination(cards: Seq[Card]): Unit = {
138     withState { state =>
139       this.gameInterface.playCombination(state.hand, state.board, cards) match {
140         case Right((updatedHand, updatedBoard)) =>
141           val updatedState = storeOpt.get.onLocalTurnStateChanged(updatedHand, updatedBoard)

144         case Left(error) => this.notifyEvent(ErrorEvent(error))

227   private def generateClientGameState(state: PlayerGameState,
228                                       turnHistory: History[PlayerGameState]): ClientGameState =
229     ClientGameState(

237   /**
238    * Execute the history method, updates the history and the resulting state
239    *
240    * @param historyUpdate History method to execute.
241    */
242   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
243     val (optState, updatedHistory) = historyUpdate()
244     this.turnHistory = updatedHistory
245     if (optState.isDefined) {
246       this.storeOpt.get.onStateChanged(optState.get)
247       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
248     }
249   }

251   /**
252    * Return if Turn is valid or not.
253    *
254    * @param currentState Current Game State.
255    * @return True if the Turn isn't valid, false anywhere.

257   private def isTurnValid(currentState: PlayerGameState) = {
258     val startState = turnHistory.headOption
259     startState.isDefined && gameInterface.validateTurn(
260       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
261   }
