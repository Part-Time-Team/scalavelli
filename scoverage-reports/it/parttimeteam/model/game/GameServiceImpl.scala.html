140   private def makeCombination(cards: Seq[Card]): Unit = {
141     withState { state =>
142       this.gameInterface.playCombination(state.hand, state.board, cards) match {
143         case Right((updatedHand, updatedBoard)) =>
144           val updatedState = storeOpt.get.onLocalTurnStateChanged(updatedHand, updatedBoard)

147         case Left(error) => this.notifyEvent(ErrorEvent(error))

153   private def pickCardsFromBoard(cards: Seq[Card]): Unit = {
