32   private val matchServerResponseListener = new MatchServerResponseListener {
33     override def gameStateUpdated(gameState: PlayerGameState): Unit = {
34 
35       storeOpt match {

37           store.onStateChanged(gameState), turnHistory)))

114

138   private def makeCombination(cards: Seq[Card]): Unit = {
139     withState { state =>
140       this.gameInterface.playCombination(state.hand, state.board, cards) match {
141         case Right((updatedHand, updatedBoard)) =>
142           val updatedState = storeOpt.get.onLocalTurnStateChanged(updatedHand, updatedBoard)
143           this.updateHistoryAndNotify(updatedState)
144 
145         case Left(error) => this.notifyEvent(ErrorEvent(error))
146       }
147 
148     }

151   private def pickCardsFromBoard(cards: Seq[Card]): Unit = {
152     withState { currentState =>
153       gameInterface.pickBoardCards(currentState.hand, currentState.board, cards) match {
154         case Right((hand, board)) => {

156           this.updateHistoryAndNotify(updatedState)

158         case Left(error) => this.notifyEvent(ErrorEvent(error))

163   private def updateCardCombination(combinationId: String, cards: Seq[Card]): Unit = {
164     withState { state =>

225   private def generateClientGameState(state: PlayerGameState,
226                                       turnHistory: History[PlayerGameState]): ClientGameState =
227     ClientGameState(

229       canUndo = turnHistory.canPrevious,

235   /**
236    * Execute the history method, updates the history and the resulting state
237    *
238    * @param historyUpdate History method to execute.
239    */
240   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
241     val (optState, updatedHistory) = historyUpdate()
242     this.turnHistory = updatedHistory
243     if (optState.isDefined) {
244       this.storeOpt.get.onStateChanged(optState.get)
245       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
246     }
247   }

249   /**
250    * Return if Turn is valid or not.
251    *
252    * @param currentState Current Game State.
253    * @return True if the Turn isn't valid, false anywhere.
254    */
255   private def isTurnValid(currentState: PlayerGameState) = {
256     val startState = turnHistory.headOption
257     startState.isDefined && gameInterface.validateTurn(
258       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
259   }
