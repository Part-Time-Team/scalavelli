187   private def generateClientGameState(state: PlayerGameState,
188                                       turnHistory: History[PlayerGameState]): ClientGameState =
189     ClientGameState(
190       playerGameState = state,
191       canUndo = turnHistory.canPrevious,
192       canRedo = turnHistory.canNext,
193       canReset = turnHistory.canPrevious,
194       canDrawCard = !isTurnValid(state)
195     )

197   /**
198    * Execute the history method, updates the history and the resulting state
199    *
200    * @param historyUpdate History method to execute.
201    */
202   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
203     val (optState, updatedHistory) = historyUpdate()
204     this.turnHistory = updatedHistory
205     if (optState.isDefined) {
206       this.storeOpt.get.onStateChanged(optState.get)
207       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
208     }
209   }

211   /**
212    * Return if Turn is valid or not.
213    *
214    * @param currentState Current Game State.
215    * @return True if the Turn isn't valid, false anywhere.
216    */
217   private def isTurnValid(currentState: PlayerGameState) = {
218     val startState = turnHistory.headOption
219     startState.isDefined && gameInterface.validateTurn(
220       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
221   }
