179   private def updateHistoryAndNotify(updatedState: PlayerGameState): Unit = {

193   private def generateClientGameState(state: PlayerGameState,
194                                       turnHistory: History[PlayerGameState]): ClientGameState =
195     ClientGameState(
196       playerGameState = state,
197       canUndo = turnHistory.canPrevious,
198       canRedo = turnHistory.canNext,
199       canReset = turnHistory.canPrevious,
200       canDrawCard = !isTurnValid(state)
201     )

203   /**
204    * Execute the history method, updates the history and the resulting state
205    *
206    * @param historyUpdate History method to execute.
207    */
208   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
209     val (optState, updatedHistory) = historyUpdate()
210     this.turnHistory = updatedHistory
211     if (optState.isDefined) {
212       this.storeOpt.get.onStateChanged(optState.get)
213       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
214     }
215   }
216 
217   /**
218    * Return if Turn is valid or not.
219    *
220    * @param currentState Current Game State.
221    * @return True if the Turn isn't valid, false anywhere.
222    */
223   private def isTurnValid(currentState: PlayerGameState) = {
224     val startState = turnHistory.headOption
225     startState.isDefined && gameInterface.validateTurn(
226       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
227   }
