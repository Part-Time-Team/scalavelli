36     override def gameStateUpdated(gameState: PlayerGameState): Unit = {
37 
38       storeOpt match {

98     withState { state =>

189   private def generateClientGameState(state: PlayerGameState,
190                                       turnHistory: History[PlayerGameState]): ClientGameState =
191     ClientGameState(
192       playerGameState = state,
193       canUndo = turnHistory.canPrevious,
194       canRedo = turnHistory.canNext,
195       canReset = turnHistory.canPrevious,
196       canDrawCard = !isTurnValid(state)
197     )

199   /**
200    * Execute the history method, updates the history and the resulting state
201    *
202    * @param historyUpdate History method to execute.
203    */
204   private def updateStateThroughHistory(historyUpdate: () => (Option[PlayerGameState], History[PlayerGameState])): Unit = {
205     val (optState, updatedHistory) = historyUpdate()
206     this.turnHistory = updatedHistory
207     if (optState.isDefined) {
208       this.storeOpt.get.onStateChanged(optState.get)
209       this.notifyEvent(StateUpdatedEvent(generateClientGameState(optState.get, turnHistory)))
210     }
211   }

213   /**
214    * Return if Turn is valid or not.
215    *
216    * @param currentState Current Game State.
217    * @return True if the Turn isn't valid, false anywhere.
218    */
219   private def isTurnValid(currentState: PlayerGameState) = {
220     val startState = turnHistory.headOption
221     startState.isDefined && gameInterface.validateTurn(
222       currentState.board, startState.get.board, currentState.hand, startState.get.hand)
223   }
