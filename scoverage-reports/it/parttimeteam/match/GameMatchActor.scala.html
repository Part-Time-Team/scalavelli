49   private def idle: Receive = {
50     case GamePlayers(players) => {

52       this.players = players
53       this.players.foreach(p => context.watch(p.actorRef))
54       require(players.size == numberOfPlayers)

60   /**
61    * Waits all players to start game
62    *
63    * @param playersReady players ready for the game
64    */
65   private def initializing(playersReady: Seq[GamePlayer]): Receive = {
66     case Ready(id, ref) => {
67       this.withPlayer(id) { p =>
68         log.info(s"player ${p.username} ready")
69 
70         val updatedReadyPlayers = playersReady :+ p.copy(actorRef = ref)
71 
72         if (updatedReadyPlayers.length == numberOfPlayers) {
73           log.info("All players ready")
74           this.initializeGame(updatedReadyPlayers)
75         } else {
76           context.become(initializing(updatedReadyPlayers) orElse terminationBeforeGameStarted())
77         }

82   /**
83    * Listen for termination messages before the match start:
84    * before all the players sent the Ready message

86   private def terminationBeforeGameStarted(): Receive = {
87     case Terminated(ref) => {
88       this.players.find(_.actorRef == ref) match {
89         case Some(player) => {
90           log.info(s"player ${player.username} terminated before the game starts")
91           // become in behaviour in cui a ogni ready che mi arriva invio il messaggio di fine partita
92           // dopo x secondi mi uccido
93           broadcastMessageToPlayers(GameEndedForPlayerLeft)

95           context.system.scheduler.scheduleOnce(20.second) {
96             log.info("Terminating game actor..")

98           }
99         }

101       }
102     }

130   /**
131    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
132    *
133    * @param playersReady all the players ready
134    */
135   private def initializeGame(playersReady: Seq[GamePlayer]): Unit = {
136     // unwatch the player with the old actor ref
137     this.players.foreach(p => context.unwatch(p.actorRef))
138 
139     this.players = playersReady
140     log.debug(s"ready players $playersReady")
141     log.debug(s"updated players $players")
142 
143     // watch the players with the new actor ref
144     this.players.foreach(p => context.watch(p.actorRef))
145 
146     this.turnManager = TurnManager[GamePlayer](players)
147     log.info("initializing game..")
148     val gameState = this.gameHelper.retrieveInitialState(players.map(p => (p.id, p.username)))
149     this.broadcastGameStateToPlayers(gameState)
150     val currentPlayer = this.turnManager.getInTurn
151     currentPlayer.actorRef ! PlayerTurn
152     this.broadcastMessageToNonCurrentPlayers(currentPlayer.id)(OpponentInTurn(currentPlayer.username))
153     context.become(inTurn(gameState, currentPlayer) orElse terminationAfterGameStarted())
154   }

156   /**
157    * Server behaviour during a player turn
158    *
159    * @param gameState    current game state
160    * @param playerInTurn current player in turn
161    */
162   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
163     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
164       log.info(s"received action $action from ${playerInTurn.username}")
165       this.gameHelper.determineNextState(gameState, playerInTurn, action) match {
166         case Right(stateResult) =>
167           this.handleStateResult(stateResult, playerInTurn)
168 
169         case Left(error) =>
170           log.error("Error resolving player action")
171           playerInTurn.actorRef ! MatchErrorOccurred(error)
172       }
173     }
174   }
175 
176   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
177 
178     // notify the state
179     this.broadcastGameStateToPlayers(stateResult.updatedState)
180 
181     // if game not ended
182     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
183 
184       playerInTurn.actorRef ! TurnEnded
185 
186       // update the turn
187       this.turnManager.nextTurn
188 
189       // notify the next player it's his turn
190       this.turnManager.getInTurn.actorRef ! PlayerTurn
191       this.broadcastMessageToNonCurrentPlayers(this.turnManager.getInTurn.id)(OpponentInTurn(this.turnManager.getInTurn.username))
192 
193       //switch the actor behaviour
194       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
195 
196     } else {
197       log.info(s"Game ended, player ${playerInTurn.username} won")
198       // game ended
199       playerInTurn.actorRef ! Won
200       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
201 
202     }
203   }
