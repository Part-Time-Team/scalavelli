18   /**
19    * Determines the next game state based on the player action
20    *
21    * @param currentState current state of the game
22    * @param playerInTurn player making the action
23    * @param playerAction action made my the player
24    * @return a state result or a string representing an error
25    */
26   def determineNextState(currentState: GameState, playerInTurn: GamePlayer, playerAction: PlayerAction): Either[MatchError, StateResult] = {
27     playerAction match {
28       case DrawCard => nextStateOnCardDrawn(currentState, playerInTurn)
29       case PlayedMove(updatedHand, updatedBoard) => nextStateOnPlayerMove(currentState, playerInTurn, updatedHand, updatedBoard)
30     }
31   }
32 
33   private def nextStateOnPlayerMove(currentState: GameState, playerInTurn: GamePlayer, updatedHand: Hand, updatedBoard: Board) = {
34     if (gameApi.validateMove(updatedBoard, updatedHand)) {
35       val updatedState = currentState
36         .getPlayer(playerInTurn.id)
37         .map(p => currentState.updatePlayer(p.copy(
38           hand = updatedHand)))
39         .get.copy(board = updatedBoard)
40 
41       Right(StateResult(
42         updatedState = updatedState,
43         additionalInformation = None
44       ))
45 
46     } else {
47       Left(MatchError.PlayerActionNotValid)
48     }
49   }
50 
51   private def nextStateOnCardDrawn(currentState: GameState, playerInTurn: GamePlayer) = {
52     val (updateDeck, cardDrawn) = gameApi.draw(currentState.deck)
53 
54     // updated player with the new card on his hand
55     val updatedState = currentState
56       .getPlayer(playerInTurn.id)
57       .map(p => currentState.updatePlayer(p.copy(
58         hand = p.hand.copy(playerCards = cardDrawn.map(_ +: p.hand.playerCards).getOrElse(p.hand.playerCards)))))
59       .get.copy(deck = updateDeck)
60 
61     Right(StateResult(
62       updatedState = updatedState,
63       additionalInformation = cardDrawn.map(CardDrawnInfo))
64     )
65   }
