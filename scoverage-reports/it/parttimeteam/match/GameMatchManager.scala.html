17   /**
18    * Determines the next game state based on the player action
19    *
20    * @param currentState current state of the game
21    * @param playerInTurn player making the action
22    * @param playerAction action made my the player
23    * @return a state result or a string representing an error
24    */
25   def determineNextState(currentState: GameState, playerInTurn: GamePlayer, playerAction: PlayerAction): Either[String, StateResult] = {
26     playerAction match {
27       case DrawCard => {
28         nextStateOnCardDrawn(currentState, playerInTurn)
29       }
30 
31       case PlayedMove(updatedHand, updatedBoard) => {
32         nextStateOnPlayerMove(currentState, playerInTurn, updatedHand, updatedBoard)
33       }
34 
35       case _ => Left("Non supported action")
36     }
37   }
38 
39   private def nextStateOnPlayerMove(currentState: GameState, playerInTurn: GamePlayer, updatedHand: Hand, updatedBoard: Board) = {
40     if (gameApi.validateMove(updatedBoard, updatedHand)) {
41       val updatedState = currentState
42         .getPlayer(playerInTurn.id)
43         .map(p => currentState.updatePlayer(p.copy(
44           hand = updatedHand)))
45         .get.copy(board = updatedBoard)
46 
47       Right(StateResult(
48         updatedState = updatedState,
49         additionalInformation = None
50       ))
51 
52     } else {
53       Left("Non valid plays")
54     }
55   }
56 
57   private def nextStateOnCardDrawn(currentState: GameState, playerInTurn: GamePlayer) = {
58     val (updateDeck, cardDrawn) = gameApi.draw(currentState.deck)
59 
60     // updated player with the new card on his hand
61     val updatedState = currentState
62       .getPlayer(playerInTurn.id)
63       .map(p => currentState.updatePlayer(p.copy(
64         hand = p.hand.copy(playerCards = cardDrawn +: p.hand.playerCards))))
65       .get.copy(deck = updateDeck)
66 
67     Right(StateResult(
68       updatedState = updatedState,
69       additionalInformation = Some(CardDrawnInfo(cardDrawn))
70     ))
71   }
