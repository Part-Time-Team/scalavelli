17   /**
18    * Determines the next game state based on the player action
19    *
20    * @param currentState current state of the game
21    * @param playerInTurn player making the action
22    * @param playerAction action made my the player
23    * @return a state result or a string representing an error
24    */
25   def determineNextState(currentState: GameState, playerInTurn: GamePlayer, playerAction: PlayerAction): Either[String, StateResult] = {
26     playerAction match {
27       case DrawCard => nextStateOnCardDrawn(currentState, playerInTurn)
28       case PlayedMove(updatedHand, updatedBoard) => nextStateOnPlayerMove(currentState, playerInTurn, updatedHand, updatedBoard)
29     }
30   }
31 
32   private def nextStateOnPlayerMove(currentState: GameState, playerInTurn: GamePlayer, updatedHand: Hand, updatedBoard: Board) = {
33     if (gameApi.validateMove(updatedBoard, updatedHand)) {
34       val updatedState = currentState
35         .getPlayer(playerInTurn.id)
36         .map(p => currentState.updatePlayer(p.copy(
37           hand = updatedHand)))
38         .get.copy(board = updatedBoard)
39 
40       Right(StateResult(
41         updatedState = updatedState,
42         additionalInformation = None
43       ))
44 
45     } else {
46       Left("Non valid plays")
47     }
48   }
49 
50   private def nextStateOnCardDrawn(currentState: GameState, playerInTurn: GamePlayer) = {
51     val (updateDeck, cardDrawn) = gameApi.draw(currentState.deck)
52 
53     // updated player with the new card on his hand
54     val updatedState = currentState
55       .getPlayer(playerInTurn.id)
56       .map(p => currentState.updatePlayer(p.copy(
57         hand = p.hand.copy(playerCards = cardDrawn +: p.hand.playerCards))))
58       .get.copy(deck = updateDeck)
59 
60     Right(StateResult(
61       updatedState = updatedState,
62       additionalInformation = Some(CardDrawnInfo(cardDrawn))
63     ))
64   }
