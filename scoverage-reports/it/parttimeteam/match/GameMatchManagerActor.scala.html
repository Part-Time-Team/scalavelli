50   private def idle: Receive = {
51     case GamePlayers(players) => {
52       this.players = players
53       require(players.size == numberOfPlayers)
54       this.broadcastMessageToPlayers(MatchFound(self))
55       context.become(initializing(Seq.empty) orElse termination())

59   /**
60    * Waits all players to start game
61    *
62    * @param playersReady players ready for the game
63    */
64   private def initializing(playersReady: Seq[GamePlayer]): Receive = {
65     case Ready(id, ref) => {
66       this.withPlayer(id) { p =>
67         log.info(s"player ${p.username} ready")
68 
69         players = players.map(p => if (p.id == id) p.copy(actorRef = ref) else p)
70         val updatedReadyPlayers = playersReady :+ p.copy(actorRef = ref)
71 
72         if (updatedReadyPlayers.length == numberOfPlayers) {
73           log.info("All players ready")
74           this.initializeGame()
75         } else {
76           context.become(initializing(updatedReadyPlayers) orElse termination())
77         }
78       }
79     }
80   }
81 
82   private def termination(): Receive = {
83     case Terminated(ref) => this.players.find(_.actorRef == ref) match {
84       case Some(player) =>
85         log.info(s"Player ${player.username} left the game")
86         this.broadcastMessageToPlayers(GameEndedForPlayerLeft)
87         self ! PoisonPill
88     }
89     case LeaveGame(playerId) => withPlayer(playerId) { player =>
90       log.info(s"Player ${player.username} left the game")
91       this.broadcastMessageToPlayers(GameEndedForPlayerLeft)
92       self ! PoisonPill
93     }
94   }
95 
96   /**
97    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
98    *
99    */
100   private def initializeGame(): Unit = {
101     this.players.foreach(p => context.watch(p.actorRef))
102     this.turnManager = TurnManager[GamePlayer](players)
103     log.info("initializing game..")
104     val gameState = this.gameMatchManager.retrieveInitialState(players.map(p => (p.id, p.username)))
105     this.broadcastGameStateToPlayers(gameState)
106     val currentPlayer = this.turnManager.getInTurn
107     currentPlayer.actorRef ! PlayerTurn
108     this.broadcastMessageToNonCurrentPlayers(currentPlayer.id)(OpponentInTurn(currentPlayer.username))
109     context.become(inTurn(gameState, currentPlayer) orElse termination())
110   }
111 
112   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
113     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
114       log.info(s"received action $action from ${playerInTurn.username}")
115       this.gameMatchManager.determineNextState(gameState, playerInTurn, action) match {
116         case Right(stateResult) =>
117           this.handleStateResult(stateResult, playerInTurn)
118 
119         case Left(errorMessage) =>
120           log.error("Error resolving player action")
121           playerInTurn.actorRef ! MatchError.PlayerActionNotValid
122       }
123 
124     }
125   }
126 
127   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
128 
129 
130     // notify the state
131     this.broadcastGameStateToPlayers(stateResult.updatedState)
132 
133     // if game not ended
134     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
135 
136       playerInTurn.actorRef ! TurnEnded
137 
138       // update the turn
139       this.turnManager.nextTurn
140 
141       // notify the next player it's his turn
142       this.turnManager.getInTurn.actorRef ! PlayerTurn
143       this.broadcastMessageToNonCurrentPlayers(this.turnManager.getInTurn.id)(OpponentInTurn(this.turnManager.getInTurn.username))
144 
145       //switch the actor behaviour
146       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
147 
148     } else {
149       log.info(s"Game ended, player ${playerInTurn.username} won")
150       // game ended
151       playerInTurn.actorRef ! Won
152       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
153 
154     }
155   }

167   private def broadcastMessageToNonCurrentPlayers(currentPlayerId: String)(message: Any): Unit = {

184       ))

188   }
