54   private def idle: Receive = {

71     case Ready(id, ref) => {

92     case Terminated(ref) => {
93       this.players.find(_.actorRef == ref) match {

135   /**
136    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
137    *
138    * @param playersReady all the players ready
139    */
140   private def initializeGame(playersReady: Seq[GamePlayer]): Unit = {
141     // unwatch the player with the old actor ref
142     this.players.foreach(p => context.unwatch(p.actorRef))
143 
144     this.players = playersReady
145     log.debug(s"ready players $playersReady")
146     log.debug(s"updated players $players")
147 
148     // watch the players with the new actor ref
149     this.players.foreach(p => context.watch(p.actorRef))
150 
151     this.turnManager = TurnManager[GamePlayer](players)
152     log.info("initializing game..")
153     val gameState = this.gameMatchManager.retrieveInitialState(players.map(p => (p.id, p.username)))
154     this.broadcastGameStateToPlayers(gameState)
155     val currentPlayer = this.turnManager.getInTurn
156     currentPlayer.actorRef ! PlayerTurn
157     this.broadcastMessageToNonCurrentPlayers(currentPlayer.id)(OpponentInTurn(currentPlayer.username))
158     context.become(inTurn(gameState, currentPlayer) orElse terminationAfterGameStarted())
159   }

161   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
162     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
163       log.info(s"received action $action from ${playerInTurn.username}")
164       this.gameMatchManager.determineNextState(gameState, playerInTurn, action) match {
165         case Right(stateResult) =>
166           this.handleStateResult(stateResult, playerInTurn)
167 
168         case Left(errorMessage) =>
169           log.error("Error resolving player action")
170           playerInTurn.actorRef ! MatchError.PlayerActionNotValid
171       }
172 
173     }
174   }
175 
176   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
177 
178 
179     // notify the state
180     this.broadcastGameStateToPlayers(stateResult.updatedState)
181 
182     // if game not ended
183     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
184 
185       playerInTurn.actorRef ! TurnEnded
186 
187       // update the turn
188       this.turnManager.nextTurn
189 
190       // notify the next player it's his turn
191       this.turnManager.getInTurn.actorRef ! PlayerTurn
192       this.broadcastMessageToNonCurrentPlayers(this.turnManager.getInTurn.id)(OpponentInTurn(this.turnManager.getInTurn.username))
193 
194       //switch the actor behaviour
195       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
196 
197     } else {
198       log.info(s"Game ended, player ${playerInTurn.username} won")
199       // game ended
200       playerInTurn.actorRef ! Won
201       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
202 
203     }
204   }
