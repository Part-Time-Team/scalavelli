51   private def idle: Receive = {
52     case GamePlayers(players) => {

60   /**
61    * Waits all players to start game
62    *
63    * @param playersReady players ready for the game
64    */
65   private def initializing(playersReady: Seq[GamePlayer]): Receive = {
66     case Ready(id, ref) => {
67       this.withPlayer(id) { p =>
68         log.info(s"player ${p.username} ready")
69 
70         players = players.map(p => if (p.id == id) p.copy(actorRef = ref) else p)
71         val updatedReadyPlayers = playersReady :+ p.copy(actorRef = ref)
72 
73         if (updatedReadyPlayers.length == numberOfPlayers) {
74           log.info("All players ready")
75           this.initializeGame()
76         } else {
77           context.become(initializing(updatedReadyPlayers) orElse termination())
78         }
79       }
80     }

83   private def termination(): Receive = {
84     case Terminated(ref) => this.players.find(_.actorRef == ref) match {
85       case Some(player) =>
86         log.info(s"Player ${player.username} left the game")

89     }
90     case LeaveGame(playerId) => withPlayer(playerId) { player =>
91       log.info(s"Player ${player.username} left the game")
92       this.broadcastMessageToPlayers(GameEndedForPlayerLeft)

97   /**
98    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
99    *
100    */
101   private def initializeGame(): Unit = {
102     this.players.foreach(p => context.watch(p.actorRef))
103     this.turnManager = TurnManager[GamePlayer](players)
104     log.info("initializing game..")
105     val gameState = this.gameMatchManager.retrieveInitialState(players.map(p => (p.id, p.username)))
106     this.broadcastGameStateToPlayers(gameState)
107     val currentPlayer = this.turnManager.getInTurn
108     currentPlayer.actorRef ! PlayerTurn
109     this.broadcastMessageToNonCurrentPlayers(currentPlayer.id)(OpponentInTurn(currentPlayer.username))
110     context.become(inTurn(gameState, currentPlayer) orElse termination())
111   }
112 
113   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
114     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
115       log.info(s"received action $action from ${playerInTurn.username}")
116       this.gameMatchManager.determineNextState(gameState, playerInTurn, action) match {
117         case Right(stateResult) =>
118           this.handleStateResult(stateResult, playerInTurn)
119 
120         case Left(errorMessage) =>
121           log.error("Error resolving player action")
122           playerInTurn.actorRef ! PlayerActionNotValidError(errorMessage)
123       }
124 
125     }
126   }
127 
128   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
129 
130 
131     // notify the state
132     this.broadcastGameStateToPlayers(stateResult.updatedState)
133 
134     // if game not ended
135     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
136 
137       playerInTurn.actorRef ! TurnEnded
138 
139       // update the turn
140       this.turnManager.nextTurn
141 
142       // notify the next player it's his turn
143       this.turnManager.getInTurn.actorRef ! PlayerTurn
144       this.broadcastMessageToNonCurrentPlayers(this.turnManager.getInTurn.id)(OpponentInTurn(this.turnManager.getInTurn.username))
145 
146       //switch the actor behaviour
147       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
148 
149     } else {
150       log.info(s"Game ended, player ${playerInTurn.username} won")
151       // game ended
152       playerInTurn.actorRef ! Won
153       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
154 
155     }
156   }
