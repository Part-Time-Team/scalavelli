61   /**
62    * Waits all players to start game
63    *
64    * @param playersReady players ready for the game
65    */
66   private def initializing(playersReady: Seq[GamePlayer]): Receive = {
67     case Ready(id, ref) => {
68       this.withPlayer(id) { p =>
69         log.debug(s"player ${p.username} ready")
70 
71         players = players.map(p => if (p.id == id) p.copy(actorRef = ref) else p)
72         val updatedReadyPlayers = playersReady :+ p.copy(actorRef = ref)
73 
74         if (updatedReadyPlayers.length == numberOfPlayers) {
75           log.debug("All players ready")
76           this.initializeGame()
77         } else {
78           context.become(initializing(updatedReadyPlayers) orElse termination())

81     }

84   private def termination(): Receive = {
85     case Terminated(ref) => this.players.find(_.actorRef == ref) match {
86       case Some(player) =>
87         log.debug(s"Player ${player.username} left the game")
88         this.broadcastMessageToPlayers(GameEndedForPlayerLeft)

92       log.debug(s"Player ${player.username} left the game")

98   /**
99    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
100    *
101    */
102   private def initializeGame(): Unit = {
103     this.turnManager = TurnManager[GamePlayer](players)
104     log.debug("initializing game..")
105     val gameState = this.gameMatchManager.retrieveInitialState(players.map(p => (p.id, p.username)))
106     this.broadcastGameStateToPlayers(gameState)
107     this.turnManager.getInTurn.actorRef ! PlayerTurn
108     context.become(inTurn(gameState, this.turnManager.getInTurn) orElse termination())
109   }

111   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
112     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
113 
114       this.gameMatchManager.determineNextState(gameState, playerInTurn, action) match {
115         case Right(stateResult) =>
116           this.handleStateResult(stateResult, playerInTurn)
117 
118         case Left(errorMessage) =>
119           log.error("Error resolving player action")
120           playerInTurn.actorRef ! PlayerActionNotValidError(errorMessage)
121       }
122 
123     }
124   }
125 
126   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
127 
128 
129     // notify the state
130     this.broadcastGameStateToPlayers(stateResult.updatedState)
131 
132     // if game not ended
133     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
134 
135       stateResult.additionalInformation match {
136         case Some(CardDrawnInfo(card)) => playerInTurn.actorRef ! CardDrawn(card)
137         case _ => playerInTurn.actorRef ! TurnEnded
138       }
139 
140       // update the turn
141       this.turnManager.nextTurn
142 
143       // notify the next player it's his turn
144       this.turnManager.getInTurn.actorRef ! PlayerTurn
145 
146       //switch the actor behaviour
147       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
148 
149     } else {
150       log.debug(s"Game ended, player ${playerInTurn.username} won")
151       // game ended
152       playerInTurn.actorRef ! Won
153       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
154 
155     }
156   }
