60   /**
61    * Waits all players to start game
62    *
63    * @param playersReady players ready for the game
64    */
65   private def initializing(playersReady: Seq[GamePlayer]): Receive = {
66     case Ready(id, ref) => {
67       this.withPlayer(id) { p =>
68         log.info(s"player ${p.username} ready")
69 
70         players = players.map(p => if (p.id == id) p.copy(actorRef = ref) else p)
71         val updatedReadyPlayers = playersReady :+ p.copy(actorRef = ref)
72 
73         if (updatedReadyPlayers.length == numberOfPlayers) {
74           log.info("All players ready")
75           this.initializeGame()

77           context.become(initializing(updatedReadyPlayers) orElse termination())

83   private def termination(): Receive = {
84     case Terminated(ref) => this.players.find(_.actorRef == ref) match {
85       case Some(player) =>
86         log.info(s"Player ${player.username} left the game")

91       log.info(s"Player ${player.username} left the game")

97   /**
98    * Inizialize the game, creating the initial state, the turn manager and changing actor behaviour
99    *
100    */
101   private def initializeGame(): Unit = {
102     this.players.foreach(p => context.watch(p.actorRef))
103     this.turnManager = TurnManager[GamePlayer](players)
104     log.info("initializing game..")
105     val gameState = this.gameMatchManager.retrieveInitialState(players.map(p => (p.id, p.username)))
106     this.broadcastGameStateToPlayers(gameState)
107     this.turnManager.getInTurn.actorRef ! PlayerTurn
108     context.become(inTurn(gameState, this.turnManager.getInTurn) orElse termination())
109   }
110 
111   private def inTurn(gameState: GameState, playerInTurn: GamePlayer): Receive = {
112     case PlayerActionMade(playerId, action) if playerId == playerInTurn.id => {
113       log.info(s"received action $action from ${playerInTurn.username}")
114       this.gameMatchManager.determineNextState(gameState, playerInTurn, action) match {
115         case Right(stateResult) =>
116           this.handleStateResult(stateResult, playerInTurn)
117 
118         case Left(errorMessage) =>
119           log.error("Error resolving player action")
120           playerInTurn.actorRef ! PlayerActionNotValidError(errorMessage)
121       }
122 
123     }
124   }
125 
126   private def handleStateResult(stateResult: StateResult, playerInTurn: GamePlayer): Unit = {
127 
128 
129     // notify the state
130     this.broadcastGameStateToPlayers(stateResult.updatedState)
131 
132     // if game not ended
133     if (!stateResult.updatedState.playerWon(playerInTurn.id)) {
134 
135       playerInTurn.actorRef ! TurnEnded
136 
137       // update the turn
138       this.turnManager.nextTurn
139 
140       // notify the next player it's his turn
141       this.turnManager.getInTurn.actorRef ! PlayerTurn
142       this.broadcastMessageToNonCurrentPlayers(this.turnManager.getInTurn.id)(OpponentInTurn(this.turnManager.getInTurn.username))
143 
144       //switch the actor behaviour
145       context.become(inTurn(stateResult.updatedState, this.turnManager.getInTurn))
146 
147     } else {
148       log.info(s"Game ended, player ${playerInTurn.username} won")
149       // game ended
150       playerInTurn.actorRef ! Won
151       this.broadcastMessageToNonCurrentPlayers(playerInTurn.id)(Lost(playerInTurn.username))
152 
153     }
154   }
