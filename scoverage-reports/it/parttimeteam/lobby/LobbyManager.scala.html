50 class LobbyManagerImpl[T <: Player] extends LobbyManager[T] {
51 
52   private var playersToLobby: Map[String, LobbyType] = Map.empty
53   private var lobbies: Map[LobbyType, Lobby[T]] = Map.empty
54 
55   /**
56    * @inheritdoc
57    */
58   override def addPlayer(player: T, lobbyType: LobbyType): Unit = {
59     lobbies = lobbies.get(lobbyType) match {
60       case Some(lobby) => lobbies + (lobbyType -> lobby.addPlayer(player))
61       case None => lobbies + (lobbyType -> GameLobby(lobbyType.numberOfPlayers).addPlayer(player))
62     }
63     playersToLobby = playersToLobby + (player.id -> lobbyType)
64   }
65 
66   /**
67    * @inheritdoc
68    */
69   override def removePlayer(playerId: String): Unit = {
70     playersToLobby.get(playerId) match {
71       case Some(lobbyType) => lobbies.get(lobbyType) match {
72         case Some(lobby) => lobby.removePlayer(playerId)
73       }
74     }
75     playersToLobby = playersToLobby - playerId
76   }
77 
78   /**
79    * @inheritdoc
80    */
81   override def getLobby(lobbyType: LobbyType): Option[Lobby[T]] = lobbies.get(lobbyType)
82 
83   /**
84    * @inheritdoc
85    */
86   override def attemptExtractPlayerForMatch(lobbyType: LobbyType): Option[Seq[T]] =
87     this.getLobby(lobbyType).flatMap(lobby => {
88       val updatedLobby = lobby.extractPlayersForMatch()
89       lobbies = lobbies + (lobbyType -> updatedLobby.first)
90       updatedLobby.second
91     })
92 }
