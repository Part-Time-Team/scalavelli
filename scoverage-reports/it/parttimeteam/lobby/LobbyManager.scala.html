51 class LobbyManagerImpl[T <: Player] extends LobbyManager[T] with CustomLogger {
52 
53   private var playersToLobby: Map[String, LobbyType] = Map.empty
54   private var lobbies: Map[LobbyType, Lobby[T]] = Map.empty
55 
56   /**
57    * @inheritdoc
58    */
59   override def addPlayer(player: T, lobbyType: LobbyType): Unit = {
60     lobbies = lobbies.get(lobbyType) match {
61       case Some(lobby) => lobbies + (lobbyType -> lobby.addPlayer(player))
62       case None => lobbies + (lobbyType -> GameLobby(lobbyType.numberOfPlayers).addPlayer(player))
63     }
64     playersToLobby = playersToLobby + (player.id -> lobbyType)
65   }
66 
67   /**
68    * @inheritdoc
69    */
70   override def removePlayer(playerId: String): Unit = {
71     playersToLobby.get(playerId) match {
72       case Some(lobbyType) => {
73         lobbies.get(lobbyType) match {
74           case Some(lobby) => {
75             lobbies = lobbies + (lobbyType -> lobby.removePlayer(playerId))
76           }
77           case _ => log(s"lobby of type $lobbyType corresponding to player $playerId  not found")
78         }
79         playersToLobby = playersToLobby - playerId
80       }
81       case None => log(s"player $playerId to remove not found")
82     }
83 
84   }
85 
86   /**
87    * @inheritdoc
88    */
89   override def getLobby(lobbyType: LobbyType): Option[Lobby[T]] = lobbies.get(lobbyType)
90 
91   /**
92    * @inheritdoc
93    */
94   override def attemptExtractPlayerForMatch(lobbyType: LobbyType): Option[Seq[T]] =
95     this.getLobby(lobbyType).flatMap(lobby => {
96       val updatedLobby = lobby.extractPlayersForMatch()
97       lobbies = lobbies + (lobbyType -> updatedLobby.first)
98       updatedLobby.second
99     })
100 }
