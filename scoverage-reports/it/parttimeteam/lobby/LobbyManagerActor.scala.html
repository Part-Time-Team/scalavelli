29   override def receive: Receive = {
30     case Connect(clientRef) => {
31       log.info(s"client $clientRef is asking for a connection")
32       val clientId = generateId
33       connectedPlayers = connectedPlayers + (clientId -> clientRef)
34       context.watch(clientRef)
35       clientRef ! Connected(clientId)
36     }
37 
38     case JoinPublicLobby(clientId, username, numberOfPlayers) => {
39       log.info(s"client $clientId wants to join a public lobby")
40       this.executeOnClientRefPresent(clientId) { ref =>
41         val lobbyType = PlayerNumberLobby(numberOfPlayers)
42         this.lobbyManger.addPlayer(GamePlayer(clientId, username, ref), lobbyType)
43         ref ! UserAddedToLobby()
44         this.checkAndCreateGame(lobbyType)
45       }
46 
47     }
48     case CreatePrivateLobby(clientId, username, numberOfPlayers) => {
49       this.executeOnClientRefPresent(clientId) { ref =>
50         val lobbyType = privateLobbyService.generateNewPrivateLobby(numberOfPlayers)
51         this.lobbyManger.addPlayer(GamePlayer(clientId, username, ref), lobbyType)
52         ref ! PrivateLobbyCreated(lobbyType.lobbyId)
53       }
54     }
55     case JoinPrivateLobby(clientId, username, lobbyCode) =>
56       this.executeOnClientRefPresent(clientId) { ref =>
57         privateLobbyService.retrieveExistingLobby(lobbyCode) match {
58           case Some(lobbyType) => {
59             val player = GamePlayer(clientId, username, ref)
60             this.lobbyManger.addPlayer(player, lobbyType)
61             ref ! UserAddedToLobby()
62             this.checkAndCreateGame(lobbyType)
63           }
64           case None => ref ! LobbyError(PrivateLobbyIdNotValidError)
65         }
66       }
67 
68     case LeaveLobby(userId) => {
69       log.info(s"client $userId")
70       this.lobbyManger.removePlayer(userId)
71     }
72 
73     case Terminated(actorRef) => {
74       log.info(s"terminated $actorRef, connected players: $connectedPlayers")
75       removeClient(actorRef)
76     }
77 
78   }

80   private def checkAndCreateGame(lobbyType: LobbyType): Unit = {
81     this.lobbyManger.attemptExtractPlayerForMatch(lobbyType) match {
82       case Some(players) => this.generateAndStartGameActor(lobbyType)(players)
83       case None =>
84     }
85   }
86 
87   private def generateAndStartGameActor(lobbyType: LobbyType)(players: Seq[GamePlayer]): Unit = {
88     val gameActor = context.actorOf(GameMatchManagerActor.props(lobbyType.numberOfPlayers, new GameInterfaceImpl()))
89     players.foreach(p => {
90       context.unwatch(p.actorRef)
91       // remove player form lobby
92       this.lobbyManger.removePlayer(p.id)
93       // remove player from connected players structure
94       this.connectedPlayers = this.connectedPlayers - p.id
95     })
96     gameActor ! GamePlayers(players)
97   }
98 
99   private def getClientRef(clientId: String): Option[ActorRef] = {
100     this.connectedPlayers.get(clientId)
101   }
102 
103   private def executeOnClientRefPresent(clientId: String)(action: ActorRef => Unit): Unit = {
104     this.getClientRef(clientId) match {
105       case Some(ref) => action(ref)
106       case _ =>
107     }
108   }
109 
110   private def removeClient(actorRef: ActorRef): Unit = {
111     this.connectedPlayers.find(_._2 == actorRef) match {
112       case Some((userId, _)) => {
113         log.info(s"removing client $actorRef from lobby and connected players list")
114         context.unwatch(actorRef)
115         this.lobbyManger.removePlayer(userId)
116         this.connectedPlayers = this.connectedPlayers - userId
117         log.info(s"removed client $actorRef from lobby and connected players list")
118       }
119       case None => log.info(s"client $actorRef not found")
120     }
121   }
