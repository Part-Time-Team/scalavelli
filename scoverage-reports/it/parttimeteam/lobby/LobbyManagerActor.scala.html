36     case JoinPublicLobby(clientId, username, numberOfPlayers) => {

51     }

81   private def generateAndStartGameActor(lobbyType: LobbyType)(players: Seq[GamePlayer]): Unit = {
82     val gameActor = context.actorOf(GameMatchManagerActor.props(lobbyType.numberOfPlayers, new GameInterfaceImpl()))
83     players.foreach(p => {

85       // remove player form lobby
86       this.lobbyManger.removePlayer(p.id)
87       // remove player from connected players structure
88       this.connectedPlayers = this.connectedPlayers - p.id

91   }

101     }
102   }

104   private def removeClient(actorRef: ActorRef): Unit = {
105     this.connectedPlayers.find(_._2 == actorRef) match {
106       case Some((userId, _)) => {

111         log.debug(s"removed client $actorRef from lobby and connected players list")
112       }
113       case None => log.warning(s"client $actorRef not found")
114     }
115   }
