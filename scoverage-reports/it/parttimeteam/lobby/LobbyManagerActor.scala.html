28   override def receive: Receive = {
29     case Connect(clientRef) => {
30       log.info(s"client $clientRef is asking for a connection")
31       val clientId = generateId
32       connectedPlayers = connectedPlayers + (clientId -> clientRef)
33       context.watch(clientRef)
34       clientRef ! Connected(clientId)
35     }
36 
37     case JoinPublicLobby(clientId, username, numberOfPlayers) => {
38       log.info(s"client $clientId wants to join a public lobby")
39       this.executeOnClientRefPresent(clientId) { ref =>
40         val lobbyType = PlayerNumberLobby(numberOfPlayers)
41         this.lobbyManger.addPlayer(GamePlayer(clientId, username, ref), lobbyType)
42         ref ! UserAddedToLobby()
43         this.checkAndCreateGame(lobbyType)
44       }
45 
46     }
47     case CreatePrivateLobby(clientId, username, numberOfPlayers) => {
48       this.executeOnClientRefPresent(clientId) { ref =>
49         val lobbyType = privateLobbyService.generateNewPrivateLobby(numberOfPlayers)
50         this.lobbyManger.addPlayer(GamePlayer(clientId, username, ref), lobbyType)
51         ref ! PrivateLobbyCreated(lobbyType.lobbyId)
52       }
53     }
54     case JoinPrivateLobby(clientId, username, lobbyCode) =>
55       this.executeOnClientRefPresent(clientId) { ref =>
56         privateLobbyService.retrieveExistingLobby(lobbyCode) match {
57           case Some(lobbyType) => {
58             val player = GamePlayer(clientId, username, ref)
59             this.lobbyManger.addPlayer(player, lobbyType)
60             ref ! UserAddedToLobby()
61             this.checkAndCreateGame(lobbyType)
62           }
63           case None => ref ! LobbyErrorOccurred(PrivateLobbyIdNotValid)
64         }
65       }
66 
67     case LeaveLobby(userId) => {
68       log.info(s"client $userId")
69       this.lobbyManger.removePlayer(userId)
70     }
71 
72     case Terminated(actorRef) => {
73       log.info(s"terminated $actorRef, connected players: $connectedPlayers")
74       removeClient(actorRef)
75     }
76 
77   }

79   private def checkAndCreateGame(lobbyType: LobbyType): Unit = {
80     this.lobbyManger.attemptExtractPlayerForMatch(lobbyType) match {
81       case Some(players) => this.generateAndStartGameActor(lobbyType)(players)

83     }

86   private def generateAndStartGameActor(lobbyType: LobbyType)(players: Seq[GamePlayer]): Unit = {
87     val gameActor = context.actorOf(GameMatchActor.props(lobbyType.numberOfPlayers, new GameHelper(new GameInterfaceImpl())))
88     players.foreach(p => {
89       context.unwatch(p.actorRef)
90       // remove player form lobby
91       this.lobbyManger.removePlayer(p.id)
92       // remove player from connected players structure
93       this.connectedPlayers = this.connectedPlayers - p.id
94     })
95     gameActor ! GamePlayers(players)
96   }
97 
98   private def getClientRef(clientId: String): Option[ActorRef] = {

100   }
101 
102   private def executeOnClientRefPresent(clientId: String)(action: ActorRef => Unit): Unit = {
103     this.getClientRef(clientId) match {
104       case Some(ref) => action(ref)

106     }
107   }
108 
109   private def removeClient(actorRef: ActorRef): Unit = {
110     this.connectedPlayers.find(_._2 == actorRef) match {
111       case Some((userId, _)) => {
112         log.info(s"removing client $actorRef from lobby and connected players list")
113         context.unwatch(actorRef)
114         this.lobbyManger.removePlayer(userId)
115         this.connectedPlayers = this.connectedPlayers - userId
116         log.info(s"removed client $actorRef from lobby and connected players list")
117       }
118       case None => log.info(s"client $actorRef not found")
119     }
120   }
