53     // List where overflowaces are converted into ace

56     val containAce: Seq[Card] = convertList.filter(card => card.rank == Rank.Ace())
57     val containTwo: Boolean = convertList.exists(card => card.rank == Rank.Two())
58 
59     (containAce.size, containTwo) match {

61       case (1, false) =>
62         convertList.foldLeft(Seq.empty[Card]) {
63           (acc, card) =>

66               case _ => acc ++ (card +: Nil)

68         }

70       case (2, true) => convertList.foldLeft(Seq.empty[Card]) {

74             case _ => acc ++ (card +: Nil)

76       }

78     }
