116    * @inheritdoc

118   override def playCombination(hand: Hand,
119                                board: Board,
120                                cards: Seq[Card]): Either[String, (Hand, Board)] = {
121     val orderedCards = cards sortByRank()
122     if (this.validateCombination(orderedCards)) {
123       hand.removeCards(orderedCards).map(updateHand => (updateHand, board.putCombination(orderedCards)))
124     } else {
125       Left("Combination not valid")
126     }
127   }
128 
129   /**

131    */
132   override def putCardsInCombination(hand: Hand, board: Board, id: String, cards: Seq[Card]): (Hand, Board) = {
133     val put = hand.removeCards(cards)
134     put match {
135       case Right(value) => (value, board.putCards(id, cards))
136       case _ => (hand, board)
137     }
