43    * @param cards Cards to remove from Hand.

45    */
46   def removeCards(cards: Seq[Card]): Either[GameError, Hand] = {

48     val updateBoardCards: Seq[Card] = boardCards.filterNot(card => cards contains card)
49 
50     Either.cond(cards forall (c => this containsCards c),
51       this.copy(playerCards = updatePlayerCards, boardCards = updateBoardCards),
52       GameError.HandNotContainCard)
53   }
54
