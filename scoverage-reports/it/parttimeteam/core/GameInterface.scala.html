93   override def create(ids: Seq[(PlayerId, PlayerName)]): GameState = {
94     var deck: Deck = Deck.shuffled
95     val playerList = ids.map(i => {
96       val playerCards = deck.draw(13)
97       deck = playerCards._1
98       Player(i._2, i._1, Hand(playerCards._2.toList))
99     })
100 
101     GameState(deck, Board.empty, playerList)
102   }

123   override def validateTurn(board: Board, startBoard: Board, hand: Hand, startHand: Hand): Boolean =
124     validateMove(board, hand) &&

126       hand.boardCards.isEmpty

128   /**
129    * @inheritdoc
130    */
131   override def pickBoardCards(hand: Hand,
132                               board: Board,
133                               cards: Seq[Card]): Either[GameError, (Hand, Board)] = {
134     board.pickCards(cards).map(updatedBoard => (hand.addBoardCards(cards), updatedBoard))
135   }
136 
137   /**
138    * @inheritdoc
139    */
140   override def playCombination(hand: Hand,
141                                board: Board,
142                                cards: Seq[Card]): Either[GameError, (Hand, Board)] = {
143     val orderedCards = cards sortByRank()
144     if (this.validateCombination(orderedCards)) {
145       hand.removeCards(cards).map(updateHand => (updateHand, board.putCombination(orderedCards)))
146     } else {
147       Left(GameError.CombinationNotValid)
148     }
149   }
150 
151   /**
152    * @inheritdoc
153    */
154   override def putCardsInCombination(hand: Hand,
155                                      board: Board,
156                                      id: String,
157                                      cards: Seq[Card]): Either[GameError, (Hand, Board)] = {
158 
159     val combBoard: CardCombination = board.combinations.filter(_.id == id).head
160 
161     if (combBoard.putCards(cards).isValid) {
162       val put = hand.removeCards(cards)
163       put match {
164         case Right(value) => Right(value, board.putCards(id, cards))
165         case Left(error) => Left(error)
166       }
167     } else {
168       Left(GameError.CombinationNotValid)
169     }
170   }
