89 class GameInterfaceImpl extends GameInterface {

91    * @inheritdoc
92    */
93   override def create(ids: Seq[(PlayerId, PlayerName)]): GameState = {
94     var deck: Deck = Deck.shuffled
95     val playerList = ids.map(i => {
96       val playerCards = deck.draw(13)
97       deck = playerCards._1
98       Player(i._2, i._1, Hand(playerCards._2.toList))
99     })
100 
101     GameState(deck, Board.empty, playerList)
102   }
103 
104   /**
105    * @inheritdoc
106    */
107   override def draw(deck: Deck): (Deck, Card) = deck.draw()
108 
109   /**

112   override def validateCombination(cards: Seq[Card]): Boolean = cards.isValid
113 
114   /**
115    * @inheritdoc
116    */
117   override def validateMove(board: Board, hand: Hand): Boolean =
118     board.combinations.forall(c => validateCombination(c.cards)) && hand.boardCards.isEmpty
119 
120   /**
121    * @inheritdoc
122    */
123   override def validateTurn(board: Board, startBoard: Board, hand: Hand, startHand: Hand): Boolean =
124     validateMove(board, hand) &&
125       (hand.playerCards.size < startHand.playerCards.size) &&
126       hand.boardCards.isEmpty
127 
128   /**
129    * @inheritdoc
130    */
131   override def pickBoardCards(hand: Hand,
132                               board: Board,
133                               cards: Seq[Card]): Either[String, (Hand, Board)] = {
134     board.pickCards(cards).map(updatedBoard => (hand.addBoardCards(cards), updatedBoard))
135   }
136 
137   /**
138    * @inheritdoc
139    */
140   override def playCombination(hand: Hand,
141                                board: Board,
142                                cards: Seq[Card]): Either[String, (Hand, Board)] = {
143     val orderedCards = cards sortByRank()
144     if (this.validateCombination(orderedCards)) {
145       hand.removeCards(orderedCards).map(updateHand => (updateHand, board.putCombination(orderedCards)))
146     } else {
147       Left("Combination not valid")
148     }
149   }
150 
151   /**
152    * @inheritdoc
153    */
154   override def putCardsInCombination(hand: Hand, board: Board, id: String, cards: Seq[Card]): (Hand, Board) = {
155     val put = hand.removeCards(cards)
156     put match {
157       case Right(value) => (value, board.putCards(id, cards))
158       case _ => (hand, board)
159     }
160   }
